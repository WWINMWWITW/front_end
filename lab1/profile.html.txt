  <script>
    /* === toast notification === */
    function showToast(message) {
      const toast = document.getElementById('toast');
      toast.textContent = message;
      toast.classList.add('show');
      setTimeout(() => toast.classList.remove('show'), 3000);
    }

    /* === toggle edit form highlight === */
    function toggleEditForm() {
      const card = document.getElementById('edit-form-card');
      card.style.outline = '2px solid #5b5bd6';
      card.scrollIntoView({ behavior: 'smooth', block: 'center' });
      setTimeout(() => { card.style.outline = 'none'; }, 1500);
      document.getElementById('field-fullname').focus();
    }

    /* === form validation and submit === */
    document.getElementById('edit-profile-form').addEventListener('submit', function(e) {
      e.preventDefault();
      let valid = true;

      /* email validation */
      const email = document.getElementById('field-email');
      const emailError = document.getElementById('email-error');
      const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailPattern.test(email.value.trim())) {
        email.classList.add('error');
        emailError.style.display = 'block';
        valid = false;
      } else {
        email.classList.remove('error');
        emailError.style.display = 'none';
      }

      /* password match validation */
      const password = document.getElementById('field-password');
      const confirm = document.getElementById('field-confirm');
      const passwordError = document.getElementById('password-error');
      if (password.value && password.value !== confirm.value) {
        confirm.classList.add('error');
        passwordError.style.display = 'block';
        valid = false;
      } else {
        confirm.classList.remove('error');
        passwordError.style.display = 'none';
      }

      if (valid) {
        showToast('profile updated successfully');
      }
    });

    /* === reset form === */
    function resetForm() {
      document.getElementById('field-fullname').value = 'Olena Lysenko';
      document.getElementById('field-email').value = 'o.lysenko@example.com';
      document.getElementById('field-phone').value = '+38 067 123 4567';
      document.getElementById('field-password').value = '';
      document.getElementById('field-confirm').value = '';
      document.querySelectorAll('.form-input.error').forEach(el => el.classList.remove('error'));
      document.querySelectorAll('.form-error').forEach(el => el.style.display = 'none');
      showToast('changes discarded');
    }

    /* === delete account confirm === */
    function confirmDelete() {
      if (confirm('are you sure you want to delete your account? this cannot be undone.')) {
        showToast('account deletion requested');
      }
    }
  </script>
